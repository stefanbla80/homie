  <script>
   function updateTemperature1()
   {
      fetch( "/getdyntemp1" )
         .then( response => {
            if( !response.ok )
               throw new Error( "fetch failed" ) ;
            return response.json() ;
         } )
         .then( json => document.querySelector("#getdyntemp1").textContent = json.getdyntemp1 )
   }
   updateTemperature1() ;
   setInterval( updateTemperature1, 1000 ) ;
  </script>
